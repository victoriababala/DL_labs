# Advanced preprocessing function
def preprocess_text(text):
    # Remove URLs
    text = re.sub(r'http\S+|www\S+|https\S+', '', text, flags=re.MULTILINE)
    # Remove user mentions
    text = re.sub(r'@\w+', '', text)
    # Remove hashtags
    text = re.sub(r'#\w+', '', text)
    # Remove non-alphabetic characters
    text = re.sub(r'[^a-zA-Z\s]', '', text)
    # Convert to lowercase
    text = text.lower()
    # Tokenize
    tokens = text.split()
    # Remove stopwords and stem
    ps = PorterStemmer()
    tokens = [ps.stem(word) for word in tokens if not word in stopwords.words('english')]
    # Join back to string
    text = ' '.join(tokens)
    return text

def create_cnn_attention_model():
    # Input layer
    input_layer = Input(shape=(max_seq_length,))
    
    # Embedding layer
    embedding_dim = 100
    embedding_layer = Embedding(input_dim=max_features + 1, 
                               output_dim=embedding_dim,
                               input_length=max_seq_length)(input_layer)
    
    # Multiple convolutional layers with different filter sizes
    conv_blocks = []
    filter_sizes = [2, 3, 4, 5]
    
    for filter_size in filter_sizes:
        # Convolutional layer
        conv = Conv1D(filters=128, 
                     kernel_size=filter_size,
                     padding='same',
                     activation='relu',
                     strides=1)(embedding_layer)
        
        # Max pooling
        max_pool = MaxPooling1D(pool_size=max_seq_length - filter_size + 1,
                               padding='valid')(conv)
        
        conv_blocks.append(max_pool)
    
    # Concatenate convolutional blocks
    concatenated = Concatenate(axis=1)(conv_blocks)
    
    # Add multi-head self-attention
    attention_layer = MultiHeadAttention(
        num_heads=8, key_dim=16
    )(concatenated, concatenated)
    
    # Add normalization and concatenate with original
    attention_layer = LayerNormalization()(attention_layer + concatenated)
    
    # Flatten
    flattened = GlobalMaxPooling1D()(attention_layer)
    
    # Dense layers with batch normalization and dropout
    dense = Dense(128, activation='relu')(flattened)
    dense = BatchNormalization()(dense)
    dense = Dropout(0.5)(dense)
    
    dense = Dense(64, activation='relu')(dense)
    dense = BatchNormalization()(dense)
    dense = Dropout(0.3)(dense)
    
    # Output layer
    output_layer = Dense(1, activation='sigmoid')(dense)
    
    # Create model
    model = Model(inputs=input_layer, outputs=output_layer)
    
    # Compile model with Adam optimizer and learning rate
    model.compile(loss='binary_crossentropy',
                 optimizer=Adam(learning_rate=0.001),
                 metrics=['accuracy'])
    
    return model

fold_no = 1
for train_idx, test_idx in kf.split(X_padded, y):
    print(f"\nTraining Fold {fold_no}/{n_folds}")
    
    # Create training and testing sets
    X_train_fold, X_test_fold = X_padded[train_idx], X_padded[test_idx]
    y_train_fold, y_test_fold = y.iloc[train_idx], y.iloc[test_idx]
    
    # Create and train model
    model = create_cnn_attention_model()
    if fold_no == 1:
        print(model.summary())
    
    # Train the model
    history = model.fit(
        X_train_fold, y_train_fold,
        validation_data=(X_test_fold, y_test_fold),
        epochs=15,
        batch_size=128,
        callbacks=[early_stopping, reduce_lr],
        verbose=1
    )
    
    # Evaluate the model
    score = model.evaluate(X_test_fold, y_test_fold, verbose=0)
    print(f"Fold {fold_no} - Loss: {score[0]:.4f}, Accuracy: {score[1]:.4f}")
    
    # Save model and history
    fold_models.append(model)
    histories.append(history)
    cv_scores.append(score[1])

    y_pred_proba = model.predict(X_test_fold)

    y_tests.append(y_test_fold)

print("\n----- FEATURE IMPORTANCE ANALYSIS -----\n")

# Function to get the most important n-grams based on filter activation
def get_important_features(model, tokenizer, texts, n=20):
    # Get embedding weights
    embedding_layer = model.layers[1]
    embedding_weights = embedding_layer.get_weights()[0]
    
    # Get Conv1D layers (they're the ones that extract n-gram features)
    conv_layers = [layer for layer in model.layers if isinstance(layer, Conv1D)]
    
    # Get word index mapping
    word_index = tokenizer.word_index
    
    # Inverse mapping
    idx_word = {v: k for k, v in word_index.items()}
    
    # For each filter size, get top activating n-grams
    important_ngrams = []
    
    for i, conv_layer in enumerate(conv_layers):
        filter_size = conv_layer.kernel_size[0]
        
        # Get filter weights
        filter_weights = conv_layer.get_weights()[0]  # shape: (filter_size, embedding_dim, num_filters)
        
        # Compute the L2 norm of each filter
        filter_norms = np.sqrt(np.sum(np.square(filter_weights), axis=(0, 1)))
        
        # Get indices of top filters
        top_filter_indices = np.argsort(filter_norms)[-n:]
        
        print(f"\nTop {n} important {filter_size}-grams:")
        for idx in reversed(top_filter_indices):
            # For demonstration, get a representative n-gram from the dataset
            for text in texts[:1000]:  # Limit search to first 1000 texts for speed
                tokens = text.split()
                if len(tokens) >= filter_size:
                    for i in range(len(tokens) - filter_size + 1):
                        ngram = " ".join(tokens[i:i+filter_size])
                        important_ngrams.append((ngram, filter_size, filter_norms[idx]))
                        break
    
    # Remove duplicates and get top n
    important_ngrams = list(set(important_ngrams))
    important_ngrams = sorted(important_ngrams, key=lambda x: x[2], reverse=True)[:n]
    
    return important_ngrams

# Function to predict if a news content is fake or real
def predict_fake_news(content, model, tokenizer):
    # Preprocess
    processed_content = preprocess_text(content)
    # Tokenize
    sequence = tokenizer.texts_to_sequences([processed_content])
    # Pad
    padded_sequence = pad_sequences(sequence, maxlen=max_seq_length, padding='post', truncating='post')
    # Predict
    prediction = model.predict(padded_sequence)[0][0]
    return prediction
